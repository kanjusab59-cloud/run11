@echo off
title Chrome Launcher Client
color 0A
setlocal enabledelayedexpansion

echo.
echo ==============================================
echo Chrome Launcher Client
echo ==============================================
echo.

set "SERVER_URL=https://run11-production.up.railway.app"
set "CONFIG_FILE=%USERPROFILE%\.chrome_device.txt"

if exist "%CONFIG_FILE%" (
set /p DEVICE_ID=<"%CONFIG_FILE%"
) else (
set "DEVICE_ID=device_%RANDOM%%RANDOM%%RANDOM%"
echo !DEVICE_ID! > "%CONFIG_FILE%"
)

set "CHROME_PATH="
if exist "C:\Program Files\Google\Chrome\Application\chrome.exe" set "CHROME_PATH=C:\Program Files\Google\Chrome\Application\chrome.exe"
if exist "C:\Program Files (x86)\Google\Chrome\Application\chrome.exe" set "CHROME_PATH=C:\Program Files (x86)\Google\Chrome\Application\chrome.exe"
if exist "%LOCALAPPDATA%\Google\Chrome\Application\chrome.exe" set "CHROME_PATH=%LOCALAPPDATA%\Google\Chrome\Application\chrome.exe"

if "%CHROME_PATH%"=="" (
echo [ERROR] Chrome not found!
echo.
set /p "CHROME_PATH=Enter Chrome path: "
)

echo [OK] Chrome: %CHROME_PATH%
echo.

set "PROFILES_DIR=%LOCALAPPDATA%\Google\Chrome\User Data"
set PROFILE_COUNT=0

for /d %%D in ("%PROFILES_DIR%\*") do (
if exist "%%D\Preferences" (
set /a PROFILE_COUNT+=1
set "PROFILE_FOLDER_!PROFILE_COUNT!=%%~nxD"
set "PROFILE_NAME_!PROFILE_COUNT!=%%~nxD"
echo [!PROFILE_COUNT!] %%~nxD
)
)

echo.
echo Device ID: %DEVICE_ID%
echo Profiles Found: %PROFILE_COUNT%
echo.

echo Sending request to server...

:: Send registration
curl -X POST "%SERVER_URL%/api/register-device" -H "Content-Type: application/json" -d "{\"deviceId\":\"%DEVICE_ID%\",\"deviceName\":\"%COMPUTERNAME%\",\"profiles\":[],\"userName\":\"%USERNAME%\"}" 2>nul

:: Send request
curl -X POST "%SERVER_URL%/api/request-access" -H "Content-Type: application/json" -d "{\"deviceId\":\"%DEVICE_ID%\",\"deviceName\":\"%COMPUTERNAME%\",\"userName\":\"%USERNAME%\",\"profiles\":[]}" 2>nul

echo.
echo Request sent to admin!
echo.
echo ==============================================
echo Waiting for admin approval...
echo Keep this window open!
echo ==============================================
echo.

:wait_loop
timeout /t 5 /nobreak >nul

:: Check status using curl
for /f %%a in ('curl -s -X POST "%SERVER_URL%/api/check-status" -H "Content-Type: application/json" -d "{\"deviceId\":\"%DEVICE_ID%\"}" 2^>nul') do set "RESPONSE=%%a"

echo %RESPONSE% | findstr "approved" >nul
if not errorlevel 1 (
echo.
echo ==============================================
echo APPROVED! Ready to launch profiles
echo ==============================================
echo.
goto monitor
)

echo Waiting for admin approval...
goto wait_loop

:monitor
echo Monitoring for launch commands...
echo.

:cmd_loop
timeout /t 3 /nobreak >nul

for /f %%a in ('curl -s -X POST "%SERVER_URL%/api/get-commands" -H "Content-Type: application/json" -d "{\"deviceId\":\"%DEVICE_ID%\"}" 2^>nul') do set "CMDS=%%a"

echo %CMDS% | findstr "launch" >nul
if not errorlevel 1 (
echo Launch command received!
for /l %%i in (1,1,%PROFILE_COUNT%) do (
echo %CMDS% | findstr "%%i" >nul
if not errorlevel 1 (
echo Opening profile %%i
start "" "%CHROME_PATH%" --profile-directory="!PROFILE_FOLDER_%%i!" "https://www.facebook.com"
)
)
echo.
)

goto cmd_loop
