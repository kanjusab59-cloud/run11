📄 File 1: chrome-client.js
const axios = require('axios');
const { exec } = require('child_process');
const fs = require('fs');
const path = require('path');
const os = require('os');

const RAILWAY_URL = 'https://run11-production.up.railway.app';
const DEVICE_ID_FILE = path.join(os.homedir(), '.chrome_launcher_device_id.txt');

function getDeviceId() {
if (fs.existsSync(DEVICE_ID_FILE)) return fs.readFileSync(DEVICE_ID_FILE, 'utf8').trim();
const newId = 'device_' + Date.now() + '_' + Math.random().toString(36).substr(2, 8);
fs.writeFileSync(DEVICE_ID_FILE, newId);
return newId;
}

function getChromePath() {
const paths = [
'C:\\Program Files\\Google\\Chrome\\Application\\chrome.exe',
'C:\\Program Files (x86)\\Google\\Chrome\\Application\\chrome.exe',
process.env.LOCALAPPDATA + '\\Google\\Chrome\\Application\\chrome.exe'
];
for (const p of paths) if (fs.existsSync(p)) return p;
return null;
}

function getAllProfiles() {
const profilesDir = path.join(process.env.LOCALAPPDATA, 'Google\\Chrome\\User Data');
if (!fs.existsSync(profilesDir)) return [];
const folders = fs.readdirSync(profilesDir);
const profiles = [];
for (const folder of folders) {
const prefFile = path.join(profilesDir, folder, 'Preferences');
if (fs.existsSync(prefFile)) {
let name = folder;
try { const data = fs.readFileSync(prefFile, 'utf8'); const match = data.match(/"name":"([^"]+)"/); if (match) name = match[1]; } catch(e) {}
if (folder === 'Default') name = 'Default';
const hasPicture = fs.existsSync(path.join(profilesDir, folder, 'Google Profile Picture.png'));
profiles.push({ name, folder, hasPicture });
}
}
return profiles;
}

async function registerDevice(profiles) {
try { await axios.post(`${RAILWAY_URL}/api/register-device`, { deviceId, deviceName: os.hostname(), profiles, userName: os.userInfo().username }); } catch(e) {}
}

async function checkCommands() {
try {
const response = await axios.post(`${RAILWAY_URL}/api/get-commands`, { deviceId });
if (response.data.success && response.data.commands.length) {
const chromePath = getChromePath();
if (!chromePath) return;
for (const cmd of response.data.commands) {
if (cmd.type === 'launch') {
const profiles = getAllProfiles();
for (const num of cmd.numbers) {
if (num >= 1 && num <= profiles.length) {
exec(`start "" "${chromePath}" --profile-directory="${profiles[num-1].folder}" "https://www.facebook.com"`);
}
}
}
}
}
} catch(e) {}
}

const deviceId = getDeviceId();
console.log(`🆔 Device ID: ${deviceId}`);
console.log(`🌐 Server: ${RAILWAY_URL}`);
registerDevice(getAllProfiles());
setInterval(() => registerDevice(getAllProfiles()), 10000);
setInterval(checkCommands, 3000);

@echo off
title Chrome Launcher Client
color 0A
echo ========================================
echo Chrome Launcher Client Setup
echo ========================================
where node >nul 2>nul
if %errorlevel% neq 0 (
echo ❌ Node.js not installed! Download from https://nodejs.org/
pause
exit /b
)
echo 📦 Installing axios...
npm install axios
echo ✅ Ready!
echo.
echo 🚀 Starting Chrome Client...
node chrome-client.js
pause
