@echo off title Chrome Launcher Client color 0A setlocal enabledelayedexpansion echo. echo ============================================== echo Chrome Launcher Client v5.0 echo ============================================== echo. set "SERVER_URL=https://run11-production.up.railway.app" set "CONFIG_FILE=%USERPROFILE%\.chrome_device.txt" if exist "%CONFIG_FILE%" ( set /p DEVICE_ID=<"%CONFIG_FILE%" ) else ( set "DEVICE_ID=device_%RANDOM%%RANDOM%%RANDOM%" echo !DEVICE_ID! > "%CONFIG_FILE%" ) echo ============================================== echo YOUR DEVICE ID: %DEVICE_ID% echo ============================================== echo. echo [INFO] Copy this Device ID to the website! echo. :: Find Chrome set "CHROME_PATH=" if exist "C:\Program Files\Google\Chrome\Application\chrome.exe" set "CHROME_PATH=C:\Program Files\Google\Chrome\Application\chrome.exe" if exist "C:\Program Files (x86)\Google\Chrome\Application\chrome.exe" set "CHROME_PATH=C:\Program Files (x86)\Google\Chrome\Application\chrome.exe" if exist "%LOCALAPPDATA%\Google\Chrome\Application\chrome.exe" set "CHROME_PATH=%LOCALAPPDATA%\Google\Chrome\Application\chrome.exe" if "%CHROME_PATH%"=="" ( echo [ERROR] Chrome not found! set /p "CHROME_PATH=Enter Chrome path: " ) echo [OK] Chrome: %CHROME_PATH% echo. :: Detect profiles set "PROFILES_DIR=%LOCALAPPDATA%\Google\Chrome\User Data" set PROFILE_COUNT=0 echo Detecting Chrome profiles... echo. for /d %%D in ("%PROFILES_DIR%\*") do ( if exist "%%D\Preferences" ( set /a PROFILE_COUNT+=1 set "PROFILE_FOLDER_!PROFILE_COUNT!=%%~nxD" set "PROFILE_NAME=%%~nxD" for /f "tokens=3 delims=:" %%A in ('findstr /c:"\"name\"" "%%D\Preferences" 2^>nul') do ( set "PROFILE_NAME=%%~A" set "PROFILE_NAME=!PROFILE_NAME:"=! set "PROFILE_NAME=!PROFILE_NAME:,=! ) if "%%~nxD"=="Default" set "PROFILE_NAME=Default" set "PROFILE_NAME_!PROFILE_COUNT!=!PROFILE_NAME!" echo [!PROFILE_COUNT!] !PROFILE_NAME! ) ) echo. echo ============================================== echo Device ID: %DEVICE_ID% echo Profiles Found: %PROFILE_COUNT% echo ============================================== echo. :: Build JSON for profiles set "PROFILES_JSON=[" for /l %%i in (1,1,%PROFILE_COUNT%) do ( if %%i gtr 1 set "PROFILES_JSON=!PROFILES_JSON!," set "PROFILES_JSON=!PROFILES_JSON!{\"name\":\"!PROFILE_NAME_%%i!\",\"folder\":\"!PROFILE_FOLDER_%%i!\",\"hasPicture\":false}" ) set "PROFILES_JSON=!PROFILES_JSON!]" :: Register device with profiles echo [1/3] Registering device... curl -s -X POST "%SERVER_URL%/api/register-device" -H "Content-Type: application/json" -d "{\"deviceId\":\"%DEVICE_ID%\",\"deviceName\":\"%COMPUTERNAME%\",\"profiles\":%PROFILES_JSON%,\"userName\":\"%USERNAME%\"}" >nul 2>&1 echo [OK] Device registered! echo. :: Send request echo [2/3] Sending request to admin... curl -s -X POST "%SERVER_URL%/api/request-access" -H "Content-Type: application/json" -d "{\"deviceId\":\"%DEVICE_ID%\",\"deviceName\":\"%COMPUTERNAME%\",\"userName\":\"%USERNAME%\",\"profiles\":%PROFILES_JSON%}" >nul 2>&1 echo [OK] Request sent!
